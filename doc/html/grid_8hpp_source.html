39 class analytic_t {
40 public:
41  std::vector<real> l1, l2, linf;
42  integer nfields_;
43  template<class Arc>
44  void serialize(Arc& a, unsigned) {
45  a & nfields_;
46  l1.resize(nfields_);
47  l2.resize(nfields_);
48  linf.resize(nfields_);
49  a & l1;
50  a & l2;
51  a & linf;
52  }
53  analytic_t() {
54  nfields_ = 0;
55  }
56  analytic_t(integer nfields) {
57  nfields_ = nfields;
58  l1.resize(nfields_);
59  l2.resize(nfields_);
60  linf.resize(nfields_);
61  for (integer field = 0; field != nfields_; ++field) {
62  l1[field] = 0.0;
63  l2[field] = 0.0;
64  linf[field] = 0.0;
65  }
66  }
67  analytic_t& operator+=(const analytic_t& other) {
68  for (integer field = 0; field != nfields_; ++field) {
69  l1[field] += other.l1[field];
70  l2[field] += other.l2[field];
71  linf[field] = std::max(linf[field], other.linf[field]);
72  }
73  return *this;
74  }
75 };
