105  static inline safe_real RL_radius(safe_real q) {
106  const safe_real q13 = std::pow(q, 1.0 / 3.0);
107  const safe_real q23 = q13 * q13;
108  const safe_real n = 0.49 * q23;
109  const safe_real d = 0.6 * q23 + std::log(1.0 + q13);
110  return n / d;
111  }
